fig, ax1 = plt.subplots(figsize=(20,8))

ax1.plot(m['sum'], 'b-o')

ax2 = ax1.twinx()
ax2.plot(m['mean'], 'r-o')
ax2.plot(m['median'], 'g-o')
ax2.tick_params(axis='y')

ax1.grid()

Longest Commit Streak

In [11]:
streak_start = 1
count = 0
max_count = 0
for i in range(len(commits)):
    if commits[i] != 0:
        count += 1
    else:
        if count > max_count:
            streak_start = i-count
            max_count = count
        count = 0
        
y1, m1, d1, *_ = data.iloc[streak_start].values
y2, m2, d2, *_ = data.iloc[streak_start+max_count-1].values

print(f"Longest Continous Commit Streak: {d1}/{m1}/{y1} to {d2}/{m2}/{y2} ({max_count} days)")
print(f"({data['Count'][streak_start:streak_start+max_count-1].sum()} commits)")
